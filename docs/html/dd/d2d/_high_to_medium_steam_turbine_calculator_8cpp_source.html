51  const std::shared_ptr<Turbine> &highToLowPressureTurbine,
52  const std::shared_ptr<Turbine> &highToLowPressureTurbineIdeal,

57  const Boiler &boiler) const {
58  const std::string methodName = std::string("HighToMediumSteamTurbineCalculator::") + std::string(__func__) + ": ";
59 
60  HighToMediumSteamTurbineCalculationsDomain highToMediumSteamTurbineCalculationsDomain;
61 
62  const PressureTurbineOperation &pressureTurbineOperation = highToMediumTurbineInput.getOperationType();
63  // std::cout << methodName << "pressureTurbineOperation=" << pressureTurbineOperation << std::endl;
64 
65  switch (pressureTurbineOperation) {
66  case PressureTurbineOperation::FLOW_RANGE:
67  highToMediumSteamTurbineCalculationsDomain =
68  calcFlowRange(availableMassFlow, highToLowTurbineInput, highToLowPressureTurbine,
69  highToLowPressureTurbineIdeal, highToMediumTurbineInput, highPressureHeaderOutput,
70  mediumPressureHeaderInput, lowPressureHeaderInput, boiler);
71  break;
72  case PressureTurbineOperation::POWER_RANGE:
73  highToMediumSteamTurbineCalculationsDomain =
74  calcPowerRange(availableMassFlow, highToLowTurbineInput, highToLowPressureTurbine,
75  highToLowPressureTurbineIdeal, highToMediumTurbineInput, highPressureHeaderOutput,
76  mediumPressureHeaderInput, lowPressureHeaderInput, boiler);
77  break;
78  case PressureTurbineOperation::POWER_GENERATION:
79  highToMediumSteamTurbineCalculationsDomain =
80  calcPowerGeneration(availableMassFlow, highToLowTurbineInput, highToLowPressureTurbine,
81  highToLowPressureTurbineIdeal, highToMediumTurbineInput,
82  highPressureHeaderOutput, mediumPressureHeaderInput, lowPressureHeaderInput,
83  boiler);
84  break;
85  case PressureTurbineOperation::STEAM_FLOW:
86  highToMediumSteamTurbineCalculationsDomain =
87  calcSteamFlow(availableMassFlow, highToLowTurbineInput, highToLowPressureTurbine,
88  highToLowPressureTurbineIdeal, highToMediumTurbineInput, highPressureHeaderOutput,
89  mediumPressureHeaderInput, lowPressureHeaderInput, boiler);
90  break;
91  case PressureTurbineOperation::BALANCE_HEADER:
92  highToMediumSteamTurbineCalculationsDomain =
93  calcBalanceHeader(availableMassFlow, highToLowPressureTurbine, highToLowPressureTurbineIdeal,
94  highToMediumTurbineInput, highPressureHeaderOutput, mediumPressureHeaderInput);
95  break;
96  default:
97  std::string msg = methodName + "PressureTurbineOperation enum not handled";
98  // std::cout << msg << std::endl;
99  throw std::invalid_argument(msg);
100  }
101 
102  return highToMediumSteamTurbineCalculationsDomain;
103 }
