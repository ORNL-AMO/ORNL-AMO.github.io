114 double interpolate( vector<double> &xData, vector<double> &yData, double x, bool extrapolate ){
115  int size = xData.size();
116  int i = 0; // find left end of interval for interpolation
117  if ( x >= xData[size - 2] ){ // special case: beyond right end
118  i = size - 2;
119  }
120  else{
121  while ( x > xData[i+1] ) i++;
122  }
123  double xL = xData[i], yL = yData[i], xR = xData[i+1], yR = yData[i+1]; // points on either side (unless beyond ends)
124  if ( !extrapolate ){ // if beyond ends of array and not extrapolating
125  if ( x < xL ) yR = yL;
126  if ( x > xR ) yL = yR;
127  }
128  double dydx = ( yR - yL ) / ( xR - xL ); // gradient
129  return yL + dydx * ( x - xL ); // linear interpolation
130 }
