102 void PrvWithDesuperheating::calculateProperties() {
103  inletProperties = SteamProperties(inletPressure, quantityType, quantityValue).calculate();
104  feedwaterProperties = SteamProperties(feedwaterPressure, feedwaterQuantityType, feedwaterQuantityValue).calculate();
105  outletProperties= SteamProperties(outletPressure, SteamProperties::ThermodynamicQuantity::TEMPERATURE,
106  desuperheatingTemp).calculate();
107 
108  inletEnergyFlow = inletProperties.specificEnthalpy * inletMassFlow;
109  feedwaterMassFlow = inletMassFlow * (inletProperties.specificEnthalpy - outletProperties.specificEnthalpy)
110  / (outletProperties.specificEnthalpy - feedwaterProperties.specificEnthalpy);
111 
112  feedwaterEnergyFlow = feedwaterMassFlow * feedwaterProperties.specificEnthalpy;
113  outletMassFlow = inletMassFlow + feedwaterMassFlow;
114  outletEnergyFlow = outletMassFlow * outletProperties.specificEnthalpy;
115 }
