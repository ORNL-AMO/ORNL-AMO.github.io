16 double interpolate(vector<double> &xData, vector<double> &yData, double x, bool extrapolate)
17 {
18  int size = xData.size();
19  int i = 0; // find left end of interval for interpolation
20  if (x >= xData[size - 2])
21  { // special case: beyond right end
22  i = size - 2;
23  }
24  else
25  {
26  while (x > xData[i + 1])
27  i++;
28  }
29  double xL = xData[i], yL = yData[i], xR = xData[i + 1], yR = yData[i + 1]; // points on either side (unless beyond ends)
30  if (!extrapolate)
31  { // if beyond ends of array and not extrapolating
32  if (x < xL)
33  yR = yL;
34  if (x > xR)
35  yL = yR;
36  }
37  double dydx = (yR - yL) / (xR - xL); // gradient
38  return yL + dydx * (x - xL); // linear interpolation
39 }
