67 double InsulatedPipeCalculator::insulationRecursive(InsulatedPipeInput input, double innerPipeDiameter, double insulationOuterDiameter, double surfaceTemperature, double interfaceTemperature, double heatLength, int i)
68 {
69  double filmTemperature;
70  filmTemperature = (surfaceTemperature + input.getAmbientTemperature()) / 2.0;
71 
72  //step 1: establish air properties
73  AirProperties airProperties = InsulatedPipeCalculator::calculateAirProperties(filmTemperature);
74 
75  //step 2: air film resistance
76  double insReynolds = ReynoldsNumber(insulationOuterDiameter, input.getWindVelocity(), airProperties.getKinViscosity()).calculate();
77  double insRayleigh = RayleighNumber(airProperties.getExpansionCoefficient(), surfaceTemperature, input.getAmbientTemperature(), insulationOuterDiameter, airProperties.getKinViscosity(), airProperties.getAlpha()).calculate();
78  double insHeatRadiation = RadiativeHeatTransferCoefficient(input.getJacketEmissivity(), surfaceTemperature, input.getAmbientTemperature()).calculate();
79  double insNusseltForced = NusseltNumber(insReynolds, airProperties.getPrandtl()).calculateForcedConvection();
80  double insNusseltFree = NusseltNumber(insRayleigh, airProperties.getPrandtl()).calculateFreeConvection();
81  double insNusseltCombined = NusseltNumber(insNusseltForced, insNusseltFree).calculate();
82  double insConvectionCombined = ConvectiveHeatTransferCoefficient(insNusseltCombined, airProperties.getConductivity(), insulationOuterDiameter).calculate();
83  double insHeatAir = insHeatRadiation + insConvectionCombined;
84 
85  //step 3: pipe resistance
86  double kPipe = this->propertyFit(input.getPipeMaterialCoefficients(), input.getPipeTemperature());
87  double pipeResistance = ThermalResistance(insulationOuterDiameter, input.getPipeDiameter(), innerPipeDiameter, kPipe).calculate();
88 
89  //step 4: insulation resistance
90  double insulationTemperature = (surfaceTemperature + interfaceTemperature) / 2.0;
91  double kInsulation = this->propertyFit(input.getInsulationMaterialCoefficients(), insulationTemperature);
92  double insulationResistance = ThermalResistance(insulationOuterDiameter, insulationOuterDiameter, input.getPipeDiameter(), kInsulation).calculate();
93 
94  //step 5: overall resistance
95  double overallResistance = insulationResistance + pipeResistance + (1.0 / insHeatAir);
96  double heatFlow = (input.getPipeTemperature() - input.getAmbientTemperature()) / overallResistance;
97 
98  //step 6: rewriting interfaceTemp & surfaceTemp
99  interfaceTemperature = input.getPipeTemperature() - (heatFlow * pipeResistance);
100  surfaceTemperature = interfaceTemperature - (heatFlow * insulationResistance);
101  double heatLengthNew = heatFlow * M_PI * insulationOuterDiameter;
102 
103  if (std::fabs(heatLengthNew - heatLength) < 0.0001)
104  {
105  return (heatLength + heatLengthNew) / 2.0;
106  }
107  else if (i > 30)
108  {
109  throw "ERROR : Pipe, Insulation - recursion limit exceeded";
110  }
111  else
112  {
113  i = i + 1;
114  return InsulatedPipeCalculator::insulationRecursive(input, innerPipeDiameter, insulationOuterDiameter, surfaceTemperature, interfaceTemperature, heatLengthNew, i);
115  }
116 }

207 double InsulatedPipeCalculator::propertyFit(std::vector<double> coefficients, double temp)
208 {
209  double property;
210  double order4 = coefficients.at(0) * std::pow(temp, 4);
211  double order3 = coefficients.at(1) * std::pow(temp, 3);
